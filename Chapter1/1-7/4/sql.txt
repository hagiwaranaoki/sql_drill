SELECT * FROM access_log WHERE access_date IN ('2013-01-01','2013-01-31') ORDER BY referer, ip_address DESC;

・INを使っているため、'2013-01-01'か'2013-01-31'の完全一致検索になっている
・refererのほうにDESCがなかった

正しくは
SELECT * FROM access_log WHERE access_date BETWEEN '2013-01-01' AND '2013-01-31' ORDER BY referer DESC, ip_address DESC;
