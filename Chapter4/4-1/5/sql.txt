CREATE TABLE INTO
  rental
  (
    id INT AUTO_INCREMENT, PRIMARY KEY,
    user_id CHAR(7),
    isbn CHAR(13),
    rental_date DATE,
    returned SMALLINT DEFAULT 0
  )
;

・CREATE TABLE INTOではなく、CREATE TABLE
・AUTO_INCREMENT, PRIMARY KEYの間のカンマはいらない

正しくは
CREATE TABLE
  rental
  (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id CHAR(7),
    isbn CHAR(13),
    rental_date DATE,
    returned SMALLINT DEFAULT 0
  )
;
