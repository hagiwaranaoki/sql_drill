SELECT 
  sh.s_name AS 店舗名, 
  sl.s_value AS 売上高 
FROM 
  sales AS sl
INNER JOIN 
    shop AS sh 
  ON 
    sl.s_id=sh.s_id
WHERE 
  sl.s_date='2012-12'
AND 
  sl.s_value < 
    (
      SELECT 
        AVG(s_value) 
      FROM 
        sales 
      WHERE 
        s_date = '2012-12'
    )
;
